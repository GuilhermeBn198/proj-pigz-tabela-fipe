<?php
// src/Controller/FipeController.php
namespace App\Controller;

use App\Dto\Fipe\BrandRequest;
use App\Dto\Fipe\ModelRequest;
use App\Dto\Fipe\YearRequest;
use App\Entity\Brand;
use App\Entity\Model;
use App\Entity\Year;
use App\Service\FipeService;
use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
use Symfony\Component\HttpFoundation\JsonResponse;
use Symfony\Component\HttpKernel\Attribute\MapRequestPayload;
use Symfony\Component\Routing\Annotation\Route;
use Symfony\Component\Security\Http\Attribute\IsGranted;

class FipeController extends AbstractController
{
    public function __construct(private FipeService $fipeService) {}
    
    #[Route('/api/fipe/brands', methods:['GET'])]
    public function listBrands(): JsonResponse
    {
        return $this->json($this->fipeService->listBrands(), 200, [], ['groups'=>'fipe:brand']);
    }
    
    #[Route('/api/fipe/brands', methods:['POST'])]
    #[IsGranted('ROLE_ADMIN')]
    public function createBrand(#[MapRequestPayload] BrandRequest $dto): JsonResponse
    {
        $brand = $this->fipeService->createBrand($dto);
        return $this->json($brand, 201, [], ['groups'=>'fipe:brand']);
    }
    
    #[Route('/api/fipe/brands/{id}', methods:['PUT','PATCH'])]
    #[IsGranted('ROLE_ADMIN')]
    public function updateBrand(Brand $brand, #[MapRequestPayload] BrandRequest $dto): JsonResponse
    {
        $brand = $this->fipeService->updateBrand($brand, $dto);
        return $this->json($brand, 200, [], ['groups'=>'fipe:brand']);
    }
    
    #[Route('/api/fipe/brands/{id}', methods:['DELETE'])]
    #[IsGranted('ROLE_ADMIN')]
    public function deleteBrand(Brand $brand): JsonResponse
    {
        $this->fipeService->deleteBrand($brand);
        return $this->json(null, 204);
    }
    
    // Similar para Model e Year...
}
